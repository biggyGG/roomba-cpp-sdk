197  uint8_t bytes_read(0);
198 
199  _sensor_data.lock();
200  {
201  // Mark all packets dirty
202  _flag_mask_dirty |= static_cast<uint64_t>(-1);
203 
204  // Process incoming data
205  if ( *reinterpret_cast<uint8_t *>(_parse_key) ) {
206  // QueryList() or Sensor()
207  for ( uint8_t packet_count = 1 ; packet_count <= *_parse_key ; ++packet_count ) {
208  uint8_t *data_offset(_raw_data), packet_size(1);
209 
210  // Calculate offset
211  data_offset += (PACKET_INFO[_packetIndex(_parse_key[packet_count])] >> 1);
212 
213  // Adjust size
214  if ( (PACKET_INFO[_packetIndex(_parse_key[packet_count])] & 0x01) ) {
215  packet_size = packetSize(_parse_key[packet_count]);
216  }
217 
218  // Move the packet into the data blob
219  bytes_read = _fnSerialRead(data_offset, packet_size);
220  if ( bytes_read != packet_size ) { _parse_status = SERIAL_TRANSFER_FAILURE; return; }
221 
222  // Record packet
223  flag_mask_received |= (static_cast<uint64_t>(1) << _packetIndex(_parse_key[packet_count]));
224  }
225  } else {
226  // Stream()
227  uint8_t checksum(0), external_checksum(0), preamble(0);
228  PacketId packet_id;
229 
230  // Parse header record
231  bytes_read = _fnSerialRead(&preamble, 1);
232  if ( bytes_read != 1 || preamble != 19 ) { _parse_status = FAILURE_TO_SYNC; return; }
233 
234  // Get byte count
235  bytes_read = _fnSerialRead(&preamble, 1);
236  if ( bytes_read != 1 || preamble == 0 ) { _parse_status = NO_DATA_AVAILABLE; return; }
237  checksum = preamble;
238 
239  // Process incoming data
240  for ( uint8_t byte_count = 0 ; byte_count < preamble ; ++byte_count ) {
241  uint8_t *data_offset(_raw_data), packet_size(1);
242 
243  // Read packet id
244  bytes_read = _fnSerialRead(reinterpret_cast<const uint8_t *>(&packet_id), 1);
245  if ( bytes_read != 1 ) { _parse_status = NO_DATA_AVAILABLE; return; }
246  checksum += packet_id;
247 
248  // Calculate offset
249  data_offset += (PACKET_INFO[packet_id] >> 1);
250 
251  // Adjust size
252  if ( (PACKET_INFO[packet_id] & 0x01) ) {
253  packet_size = packetSize(packet_id);
254  }
255 
256  // Move the packet into the data blob
257  bytes_read = _fnSerialRead(data_offset, packet_size);
258  if ( bytes_read != packet_size ) { _parse_status = SERIAL_TRANSFER_FAILURE; return; }
259  for ( uint8_t i = 0 ; i < packet_size ; ++i ) { checksum += data_offset[i]; }
260 
261  // Record packet
262  flag_mask_received |= (static_cast<uint64_t>(1) << _packetIndex(packet_id));
263  }
264 
265  // Test checksum
266  bytes_read = _fnSerialRead(&external_checksum, 1);
267  if ( bytes_read != 1 || static_cast<uint8_t>(checksum + external_checksum) ) { _parse_status = INVALID_CHECKSUM; return; }
268  }
269  _flag_mask_dirty &= ~flag_mask_received;
270  *reinterpret_cast<uint8_t *>(_parse_key) = 0;
271  _parse_status = SUCCESS;
272  }
273  _sensor_data.unlock();
